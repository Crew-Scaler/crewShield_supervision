### Method 3: Python requests with Validation
```python
import requests
import time
import os

output_dir = "/Users/tamnguyen/Documents/GitHub/ksi_watch/KSI-TPR-04_25-12A_SupplyChainRiskMonitoring/references/cluster_3_multi_agent_security"
arxiv_ids = [
    "2505.02077", "2507.06850", "2506.19676", "2503.12188", "2503.13657",
    "2504.07461", "2510.23883", "2508.09815", "2510.06445", "2506.04133",
    "2510.16219", "2509.01211", "2505.08807", "2510.11108", "2406.02630"
]

headers = {'User-Agent': 'Mozilla/5.0 (compatible; ArXiv-Research-Bot/1.0)'}

for arxiv_id in arxiv_ids:
    url = f"https://arxiv.org/pdf/{arxiv_id}"
    filepath = os.path.join(output_dir, f"{arxiv_id}.pdf")

    try:
        response = requests.get(url, headers=headers, timeout=15)
        response.raise_for_status()

        with open(filepath, 'wb') as f:
            f.write(response.content)

        print(f"✓ {arxiv_id}: {len(response.content)} bytes")
    except Exception as e:
        print(f"✗ {arxiv_id}: {e}")

    time.sleep(3.5)
```
